class Solution {
public:
    bool canJump(vector<int>& nums) 
    {
        int max = 0; //讽eiǐ飒撼踏jB计
        int i; //index
        for(i = 0 ; i<nums.size() ; i++)
        {
            if(i > max)//讽eiǐ撼踏jB计常ǐぃ = J0
            {
                break;
            }
            max = std::max(max,i+nums[i]);//{b害旄m+{biǐB计莪痈环
        }
        return i==nums.size();
    }
};
