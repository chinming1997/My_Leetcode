class Solution {
public:
    bool isPerfectSquare(int num)
    {
        if(num==1) return true;
        if(num < 1) return false;
        int left = 0;
        int right = num/2;
        while(left<=right)
        {
            long mid = (left+right)/2;
            long value = mid*mid;
            if(value==num) return true;
            else
            {
                if(value<num)
                {
                    left = mid+1;
                }
                else
                {
                    right = mid-1;
                }
            }
            
        }
        return false;
    }

};
